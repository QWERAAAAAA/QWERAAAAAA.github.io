##### 1.浮点数的近似存储真是害人不浅！分类平均

```C++
#include<iostream>
#include<iomanip>
using namespace std;
int main(){
    int n,k,i;	
    float a=0,b=0,A=0,B=0;	//这里设置a,b,A,B都是浮点数导致每一次加误差越来越大
    
    //int a=0,b=0,A=0,B=0; 	算都用整型算，没有误差
    //float(a)/A 后面输出时再进行转化，否则只会去整输出
    
    cin>>n>>k;
    for(i=1;i<=n;i++){
        if(i%k==0) {a+=i; A+=1;}
        else {b+=i; B+=1;}
    }
    cout<<fixed<<setprecision(1);
    cout<<a/A<<" "<<b/B<<endl;
    return 0;

}
```

##### 2. 一个for循环可以使多个变量循环哦（for循环的实际应用）数字金字塔

```C++
#include<iostream>
#include<iomanip>
using namespace std;
int main(){
    int n,a,k=1;
    cin>>n;
    a=(n+1)*n/2;
    for(int i=n;i>=1;i--){
        for(int j=1;j<=i;j++){
            cout<<setw(2)<<setfill('0')<<k;		//这里加一个变量k
            k++;  
        }
        cout<<endl;
    }
    return 0;
}
```

###### 类似处理：计数问题 

```C++
#include<iostream>
using namespace std;
int main(){
    int n,x,k,count=0;
    cin>>n>>x;
    if(n<10 && x!=0) count=1;	
    else{
        if(n<x) count=0;
        else{	
            for(int i=1;i<=n;i++){
            k=i;		// 关键一步，使得i不受改变）
            while(k>0){
                if(k%10==x) count+=1;	//不断的把拆出来的数用来判断
                k/=10;	// 不断的拆开位数
            }
            }
        }
    }
    cout<<count<<endl;
    return 0;
}
```

在for循环中多引入一个变量k；k=i；i用于循环，k用于运算；
